class FeedForward(object):
    def __init__(self):
        
        self.inputLayerSize = 2 # mme dva vstupy
        self.outputLayerSize = 1 # jeden vstup
        self.hiddenLayerSize = 3 # tri neurny v skrytej vrstve
        
        # inicializujeme vhy (2x3) kt. s medzi vstupnou a skrytou vrstvou
        self.FIRST_WEIGHTS = np.random.randn(self.inputLayerSize,self.hiddenLayerSize)
        # inicializujeme vhy (3x1) kt. s medzi skrytou a vstupnou vrstvou
        self.SECOND_WEIGHTS = np.random.randn( self.hiddenLayerSize,self.outputLayerSize)  
        
        
    def forward_function(self, X): #prechdzame ns
        
        #vynsobme prv vhy so vstupom a stame
        self.product_1 = np.dot(X,self.FIRST_WEIGHTS) 
        #zavolme aktivan funkciu
        self.activ_output = self.activation_function(self.product_1)
        #vynsobme vsledok aktiv. fun. s druhmi vhami a stame
        self.product_2 = np.dot(self.activ_output, self.SECOND_WEIGHTS)
        #opt zavolme aktiv. fun. a vrtime vsledok ns
        return self.activation_function(self.product_2) 

    def activation_function(self, s): #aktivan funkcia typu sigmoid
        return 1 / (1 + np.exp(-s))

    def act_func_derivation(self, s): #derivcia aktivanej funkcie
        return s * (1 - s)

    def back_propagation(self, X, y, nn_output): #metoda sptnho renia chb
        #urme chybu vsledku ns
        self.output_error = y - nn_output  
        #vypotame o kolko upravme druh vhy 
        self.output_delta = self.output_error * self.act_func_derivation(nn_output)  
        #urme ako moc prispeli druh vhy k chybe
        self.product_2_error = self.output_delta.dot(self.SECOND_WEIGHTS.T)
        #vypotame o kolko upravme prv vhy 
        self.product_2_delta = self.product_2_error * self.act_func_derivation(self.activ_output)  
        

        #upravme prv vhy
        self.FIRST_WEIGHTS += X.T.dot(self.product_2_delta)  
        #upravme druh vhy
        self.SECOND_WEIGHTS += self.activ_output.T.dot(self.output_delta)  
        
    def train(self, X, y):#trenovanie ns
        self.back_propagation(X, y, self.forward_function(X))
